{

	while (y != '#')
	{

		if (x == 18)
		{
			printf("\n Digite o caractere:");

		//de char -> int 
		scanf(" %c", &y);

		//number

		{
			printf("%c = um número");
		}

		//simbolo x!= 36
		if (x >= 33 && x <= 47 || x >= 58 && x <= 64)
		{
			printf("%c = um símbolo\n", y);
		}

		if (x >= 65 && x <= 90)
		{
			//vogal
			if (x == 65 || x == 69 || x == 73 || x == 79 || x == 85)
			{

			}
			else
			{
				printf("%c = uma consoante em caixa-alta \n", y);
			}
		}

		//letras 
		if (x >= 97 && x <= 122)

			//vogal

			{
				printf("%c = uma vogal \n", y);
